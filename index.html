class ChatPaperSearcher:
    """
    完整工作流（全部在 chatpaper.com/zh-CN 上完成）：
      1. 打开 chatpaper.com → 搜索框输入关键词 → 搜索
      2. 点击 Sort by: Published Date
      3. 遍历搜索结果
      4. 进详情页 → 页面文本统计关键词次数（不下载 PDF）
      5. 关键词 ≥ 阈值 → 提取信息 → 录入飞书
    """

    def search_keyword(self, keyword):
        """在 ChatPaper 搜索一个关键词"""
        self.page.goto("https://chatpaper.com/zh-CN")

        # 找到搜索框，输入关键词
        search_input = self._find_element([...])
        search_input.fill(keyword)

        # 点击搜索按钮
        self._click_search()

        # 切换排序为 Published Date
        sort_btn = self._find_element(['text="Published Date"', ...])
        sort_btn.click()

        # 提取搜索结果
        return self._extract_results()

    def check_keywords_on_page(self, url):
        """
        打开论文详情页，在网页文本中统计关键词。
        不下载 PDF，直接在页面上检索。
        """
        detail = self.context.new_page()
        detail.goto(url, wait_until="networkidle")

        # 获取页面全部文本
        page_text = detail.inner_text("body").lower()

        # 统计每个关键词出现次数
        total = 0
        for kw in CONFIG["page_check_keywords"]:
            total += page_text.count(kw.lower())

        detail.close()
        return total

    def search_all_keywords(self):
        """搜索所有关键词 → 合并去重 → 逐篇检查"""
        self.start_browser()
        try:
            for kw in CONFIG["search_keywords"]:
                results = self.search_keyword(kw)

                paper.keyword_count = self.check_keywords_on_page(
                    paper.chatpaper_url
                )
                # 符合条件的提取详细信息...
        finally:
            self.close_browser()

⚙️ config.yaml
# 数据源: https://chatpaper.com/zh-CN

# 在 ChatPaper 搜索框中依次搜索的关键词
search_keywords:
  - "GUI Agent"
  - "CUA"
  - "Web Agent"
  - "mobile agent"
  - "computer use"

# 在论文详情页上统计出现次数的关键词（不下载 PDF）
page_check_keywords:
  - "web agent"
  - "gui agent"
  - "computer use"
  - "mobile agent"
  - "CUA"

keyword_threshold: 2    # ≥ 此值录入飞书
days_lookback: 1        # 最近 N 天

headless: true          # 无头模式（首次调试改 false）
slow_mo: 300            # 操作间隔(ms)
⚙️ .github/workflows/paper_track.yml
name: 📄 Paper Tracker — Daily Run
on:
  schedule:
    - cron: '0 1 * * *'   # UTC 01:00 = 北京 09:00
  workflow_dispatch:

jobs:
  track-papers:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with: { python-version: '3.11', cache: 'pip' }
      - run: |
          pip install -r requirements.txt
          playwright install chromium
          playwright install-deps chromium
      - run: python paper_tracker.py
        env:
          FEISHU_APP_ID: ${{ secrets.FEISHU_APP_ID }}
          FEISHU_APP_SECRET: ${{ secrets.FEISHU_APP_SECRET }}
          FEISHU_WIKI_TOKEN: ${{ secrets.FEISHU_WIKI_TOKEN }}
      - run: |
          git config user.email "action@github.com"
          git config user.name "Paper Tracker Bot"
          git add processed_papers.json || true
          git diff --staged --quiet || git commit -m "📄 [$(date +%Y-%m-%d)]"
          git push || true
